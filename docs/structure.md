│
├── backend/

│   └── go/                     # Go components
│
├── frontend/                   # UI
│
├── docs/                       # Documentation
│   ├── api/
│   ├── architecture.md
│   └── structure.md
│
├── scripts/                    # DevOps scripts
│   ├── build/                  # build scripts
│   ├── deploy/                 # deploy scripts
│   └── testing/                # testing scripts
│
├── tests/
│   ├── unit/                   # Unit tests
│   ├── integration/            # Integration tests

│
├── deploy/
│   ├── docker/
│   └── k8s/
│
├── .gitignore
├── LICENSE
